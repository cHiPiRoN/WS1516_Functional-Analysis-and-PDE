	\begin{proposition}\label{c2seXthm9}
		Let $U\subset\mathbb{R}^n$ open and let $(K_i)_{i\in\mathbb{N}}$ family of compact sets $K_i\subset\mathbb{R}^n$, s.t.
		\[
			K_i\subseteq K_{i+1},\quad\forall i
		\]
		with
		\[
			U = \bigcup_{i=0}^\infty K_i.
		\]
		We assume, that $\forall x\in U\exists i_k\in\mathbb{N},r_k>0$ s.t. $B(x,r_k)\subset K_{i_k},\,(\forall i\geq ik$).
		We define for any $f\in C(U;Y)$
		\[
			\|f\|_i \coloneqq \sup_{x\in K_i} \|f(x)\| <\infty,\quad\text{ since }K_i\text{ compact},\quad\forall i\in\mathbb{N}.
		\]
		Moreover, $\forall f,g\subset C(U;Y)$ let
		\[
			d(f,g) \coloneqq \sum_{i=1}^\infty 2^{-1}\underbrace{\frac{\|f-g\|_i}{1+\|f-g\|_i}}_{\le 1} \leq \sum_{i=1}^\infty 2^{-1} = 1.
		\]
		\underline{Then: } $\|\cdot\|_i$ is a \underline{seminorm} on $C(U;Y),\,\forall i$ and $d$ ia a \underline{metric} on $C(U;Y)$.
	\end{proposition}
	\begin{remark}
		Above metric $d$ is called the \emph{Fréchet metric}, generated by the family of seminorms $(\|\cdot\|_i)_{i=1}^\infty$.
	\end{remark}

	\begin{proposition}\label{c2seXthm10}
		Let $f:\mathbb{N}\to C(U;Y)$ be a sequence of fuctions in $C(U;Y)$, and let $U$ as in \cref{c2seXthm9}. The following statements are equivalent
		\begin{itemize}
			\item [$(a)$ ] $g\in C(U;Y)$ s.t. $\lim_{n\to\infty} d(f_n,g)\to 0$ (i.e. $f_n\to g$ in $d$).
			\item [$(b)$ ] $\forall i\geq 1:\,\lim_{n\to\infty}\|f_n-g\|_i = 0$.
			\item [$(c)$ ] $\forall K\subset U$ compact, it holds that $\sup_{x\in K}\|f_n(x)-g(x)\|\overset{n\to\infty}\to 0$.
		\end{itemize}
	\end{proposition}

	\begin{proof}
		\textbf{$(a)\Rightarrow (b)$ }
		\[
			d(f_n,g) = \sum_{i=1}^\infty 2^{-1}\frac{\|f_n-g\|_i}{1+\|f_n-g\|_i} \Rightarrow \bigg[ d\to 0 \Rightarrow \|f_n-g\|_i\to 0,\quad\forall i \bigg].
		\]
		\textbf{$(b)\Rightarrow (a)$}
		\[
			\|\cdot\|_i\to 0,\quad\forall i
		\]
		implies that
		\[
			\forall \varepsilon>0\exists i_0(\varepsilon)\text{ s.t. } \sum_{i\geq i_0}^\infty 2^{-2}\frac{\|\cdot\|_i}{1+\|\cdot\|_i} \leq \sum_{i\geq i_0}^\infty 2^{-1} < \varepsilon,
		\]
		and this implies
		\[
			d(f_n,g) \leq \sum_{i=1}^{i_0-1} 2^{-1} \frac{\|f_n-g\|_i}{1+\|f_n-g\|_i} + \varepsilon
		\]
		so there is $n$ s.t. $\|f_n-g\|_i<\varepsilon,\quad\forall n\geq n_0,\quad\forall i\in[1,..,i_0]$.
		Therefore $d(f_n,g) \leq 2\varepsilon,\quad\forall n\geq n_0$, so $d\to 0$.\newline\newline\noindent
		\textbf{$(b)\Leftarrow (c)$ } Easy.\newline\newline\noindent
		\textbf{$(b)\Rightarrow (c)$ } $K\subset U$ compact, therefore $\forall x\in K\exists i_k\in\mathbb{N}$ and $r_x>0$ s.t.
		\[
			B(x,r_x)\subset K_{i_x}.
		\] 
		$K\subset\bigcup_{x\in K} B(x,r_x)$ and as $K$ compact there are $x_1,...,x_q$ with
		\[
			K\subset\bigcup_{j=1}^q B(x_j,r_{x_j})\subset \bigcup_{j=1}^q K_{i_{x_j}}\subset K_{\bar{i}},
		\]
		where $\bar{i} = \max_{j=1,...,k}\{i_{x_j}\}$. Then
		\[
			\|f_n-g\|_i\to 0\,\Rightarrow\, \sup_{x\in K} \|f_n(x)-g(x)\|\to 0.
		\]
	\end{proof}

\section{Differentiable functions}\label{ce2se3}
	
	\begin{definition}\label{c2se3def12} Let $U\subset\mathbb{R}^n$ open. Then we define
	\[
		C^k(U;\mathbb{R}^m) \coloneqq  \{f:U\to\mathbb{R}^m|\,f\,k\text{ times differentiable and }D^lf\text{ continuous},\,\forall 0\leq l\leq k \}
	\] 
	\[
		C^\infty(U,\mathbb{R}^m) \coloneqq \{f:U\to\mathbb{R}^m|\,f\textbf{ smooth}\}.
	\]
	\end{definition}
	\begin{remark}
		For $l=2$:
		\[
			D^lf(x) = \left\{\frac{\partial}{\partial x_{j_1}}\frac{\partial}{\partial x_{j_2}} f\right\}_{j_1,j_2=1}^n = \partial_{x_1}^{\alpha_1}\partial_{x_2}^{\alpha_2}...\partial^{\alpha_n}_{x_n}f = \partial^\alpha f,
		\]
		with the multi-index $\alpha$ satisfying $|\alpha| = \sum_{j=1}^n\alpha_j = 2$, and $\alpha_j\in\mathbb{N},\,\forall j$.
	\end{remark}

	$D^lf(x)$ is an $l$-multilinear form ( $=$ matrix is a tensor ).\newline\noindent
	On a finite dimensional space, all norms are equivalent.\newline\noindent
	Euclidean norm: $\|D^lf(x)\| = \sqrt{\sum_{\alpha,|\alpha|=l} \|\partial^\alpha f(x)\|^2}$. If the space is $\mathbb{R}^n$, the norm is often just written as $|\cdot|$.\newline\noindent
	$C^k\subset C(U;\mathbb{R}^m) \Rightarrow $ Use the metric $d$? (Yes, if $U$ is bounded!) However, nothing guarantees the completeness of the space $C^k$ with this metric $d$. We need to define a more suitable norm. If $U$ is bounded, then $\overline{U}$ is compact. We restrict ourselves to functions that have a ``nice'' behaviour on $\overline{U}$, i.e. that do not explode on the boundary of $U$.

	\begin{definition}\label{c2seXdef13}
		Let $U\subset\mathbb{R}^n$ open \underline{and} bounded. Define
		\[
			C^k(\overline{U};\mathbb{R}^m) \coloneqq \{ f\in C^k(U;\mathbb{R}^m)|\,D^lf\text{ has a continuous extension to }\overline{U}\}
		\]
		We do the same with smooth functions:
		\[
			C^\infty(\overline{U};\mathbb{R}^m) \coloneqq \bigcap_{k=c}^\infty C^k(\overline{U};\mathbb{R}^m)
		\]
	\end{definition}
	\begin{remark}
		If $\overline{U}$ is closed and bounded in $\mathbb{R}^n$, then $\overline{U}$ is compact, so
		\[
			\sup_{x\in U}\|D^lf(x)\|\le\infty,\quad\forall 0\leq l\leq k
		\]
		Using those definitions above, we can define norms on the spaces.
	\end{remark}

	\begin{definition}\label{c2seXdef14}
		We define
		\[
			[f]_{C^l} \coloneqq \sup_{x\in\overline{U}} \|D^lf(x)\|,\quad 0\leq l\leq k,\quad f\in C^k(\overline{U};\mathbb{R}^m)
		\]
		As $f\in C^k(\overline{U};\mathbb{R}^m)$ has a continuous extension on $\overline{U}$, and $\overline{U}$ is compact, the supremum exists. Therefore
		\[
			[f]_{C^k}<\infty.
		\]
		Next, we define
		\[
			\|f\|_{C^k} \coloneqq \sum_{l=0}^k [f]_{C^l}
		\]
		and
		\[
			d(f,g) \coloneqq \sum_{i=0}^\infty 2^{-i} \frac{[f-g]_{C^i}}{1+[f-g]_i}.
		\]
		This metric is welldefined for all $f,g\in C^\infty(\overline{U};\mathbb{R}^m)$.
	\end{definition}

	\begin{theorem}\label{c2seXthm15}
		For all $0\leq l\leq k$ we have that $[\cdot]_{C^l}$ is a seminorm on $C^k(\overline{U};\mathbb{R}^m)$. Moreover, $\|\cdot\|_{C^k}$ is a norm on $C^k(\overline{U};\mathbb{R}^m)$ and $d(\cdot,\cdot)$ is a metric on $C^\infty(\overline{U};\mathbb{R}^m)$.\newline\noindent
		In this case, we have that $Y=\mathbb{R}^m$, and this space is complete and separable. Therefore $(C^k(\overline{U},\mathbb{R}^m),\|\cdot\|_{C^k})$ is a separable Banach space. Analogeously, the space $(C^\infty(\overline{U};\mathbb{R}^m),d)$ is a complete, separable, metric space.
	\end{theorem}

	\begin{remark} Consider a bounded subset $U\subset\mathbb{R}^m$. Then we can always construct a sequence $(K_i)_{i\in\mathbb{N}}$ of compact subsets, s.t.
	\[
		\bigcup_i K_i = U.
	\]
	Thus every $f\in C^k(U;\mathbb{R}^m)$ when restricted to a compact set, i.e. $f_{|K_i}$, implies that we can define $\|f_{|K_i}\|_{C^k}$, as $f\in C^k(K_i;\mathbb{R}^m)$. Using this, we can define a metric on $C^k(U;\mathbb{R}^m)$ (that is \underline{NOT} a norm).

	Next, we will talk about the \emph{compact support} of a function.

	\begin{definition}\label{c2seXdef16}
		Let $(X,\mathcal{T}),(Y,\|\cdot\|)$, and let $f:X\to Y$. The \emph{support of } $f$ is defined as
		\[
			\supp\,f \coloneqq \overline{\{x\in X|\,f(x)\not=0\}}.
		\]
		The emoh{space of continuous functions with compact support} is then defined as
		\[
			C_c(X;Y) \coloneqq \{f\in C(X;Y)|\,\supp f\text{ is compact}\}.
		\]
		For all $U\subset\mathbb{R}^n$ that are open, and $Y=\mathbb{R}^m$, we define
		\[
			C_c^k(U;\mathbb{R}^m) \coloneqq \{f\in C^k(U;\mathbb{R}^m)|\,\supp f\text{ compact} \},
		\]
		and
		\[
			\mathcal{D}(U;\mathbb{R}^m) \coloneqq C^\infty_c(U;\mathbb{R}^m) = C^\infty(U;\mathbb{R}^m)\cap C_c(U;\mathbb{R}^m).
		\]
	\end{definition}
	In many cases the notion $C^k_0$ instead of $C_c^k$ is used. \newline\noindent
	If $Y=\mathbb{R}$, then we write $C^k(U), C^k_c(U)$ instead of $C^k(U;\mathbb{R}), C^k_c(U;\mathbb{R})$.\newline\noindent
	For $U\subset\mathbb{R}^n$, we have that $\supp f$ is compact $\Leftrightarrow$ $\supp f$ is bounded.\newline\noindent
	\begin{example}
		Let $f:[0,\infty)\to\mathbb{R}$ defined as
		\[
			f(x) = \begin{cases} 1-x&0\leq x\leq 1\\0&x\geq 1\end{cases}
		\]
		Then 
		\begin{IEEEeqnarray}{rCl}
			f &\in& C_c([0,\infty),\mathbb{R});\\
			f &\in& C_c([0,\infty],\mathbb{R});\\
			f &\not\in& C_c((0,1),\mathbb{R}).
		\end{IEEEeqnarray}
	\end{example}

	Let's talk about weaker continuity.

	\begin{definition}[Hölder continuity]\label{c2seXdef17}
		Let $\alpha\in(0,1]$, and let $A\subset\mathbb{R}^n$. We say that $f:A\to Y$ is \emph{Hölder continuous with parameter $\alpha$}, if 
		\[
			[f]_\alpha \coloneqq \sup_{x,y\in A,x\not=y} \frac{\|f(x)-f(y)\|}{\|x-y\|^\alpha} < \infty.
		\]
	\end{definition}
	\begin{remark}
		$f$ is continuous in the classical sense, but not necessary differentiable. An easy counterexample can be constructed using the square-root, as
		\[
			\frac{\sqrt{x}}{x^\alpha} < \infty \Rightarrow \alpha =\frac{1}{2}.
		\]
	\end{remark}

	\begin{definition}\label{c2seXdef18}
		Let $\alpha\in(0,1]$, and $A\subset\mathbb{R}^n$. We define
		\[
			C^0,\alpha(A;\mathbb{R}^m) \coloneqq \{f\in C_b(A;\mathbb{R}^m)|\,\underbrace{f\text{ is }\alpha\text{-Hölder continuous}}_{[f]_\alpha < \infty}\},
		\]
		and let
		\[
			\|f\|_\alpha \coloneqq \sup_{x\in A} \|f(x)\| + \underbrace{[f]_{\alpha}}_{\text{Quasiderivative}}.
		\]
		Let now $U\subset\mathbb{R}^n$ open and bounded. Define
		\[
			C^{k,\alpha}(U;\mathbb{R}^m) \coloneqq \{f\in C^k(\overline{U};\mathbb{R}^m)|\,[D^kf]_\alpha < \infty\}
		\]
		(those functions have normal derivatives up to order $k$, and this $k$-th derivative is nearly again differentiable, but in fact just the quasiderivative exists).
		On this space define a norm
		\[
			\|f\|_{k,\alpha} \coloneqq \|f\|_k + \underbrace{[D^kf]_\alpha}_{D^kf\text{ is `almost' diff.}}
		\]
	\end{definition}

	\begin{proposition}\label{c2seXthm19}
		$[\cdot]_\alpha$ is a seminorm on $C^{0,\alpha}(A;\mathbb{R}^m)$.\newline\noindent
		$\|\cdot\|_\alpha$ is a norm on $C^{0,\alpha}(A;\mathbb{R}^m)$.\newline\noindent
		$\mathbb{R}^m$ is a complete space, so $(C^{0,\alpha}(A;\mathbb{R}^m),\|\cdot\|_\alpha)$ is a Banach space.\newline\noindent
		$\|\cdot\|_{k,\alpha}$ is a norm on $C^{k,\alpha}(U;\mathbb{R}^m)$ and $(C^{k,\alpha}(U;\mathbb{R}^m),\|\cdot\|_{k,\alpha})$ is a Banach space. 
	\end{proposition}

	\begin{remark}
		$C^{0,\alpha}(A;\mathbb{R}^m)$ is not separable, unless $A$ is finite.
	\end{remark}
